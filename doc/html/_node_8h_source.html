34  template <void A(Socket *s), bool TIMER>
35  static void accept_cb(ListenSocket *listenSocket) {
36  uv_os_sock_t serverFd = listenSocket->getFd();
37  Context *netContext = listenSocket->nodeData->netContext;
38  uv_os_sock_t clientFd = netContext->acceptSocket(serverFd);
39  if (clientFd == INVALID_SOCKET) {
40  /*
41  * If accept is failing, the pending connection won't be removed and the
42  * polling will cause the server to spin, using 100% cpu. Switch to a timer
43  * event instead to avoid this.
44  */
45  if (!TIMER && !netContext->wouldBlock()) {
46  listenSocket->stop(listenSocket->nodeData->loop);
47 
48  listenSocket->timer = new Timer(listenSocket->nodeData->loop);
49  listenSocket->timer->setData(listenSocket);
50  listenSocket->timer->start(accept_timer_cb<A>, 1000, 1000);
51  }
52  return;
53  } else if (TIMER) {
54  listenSocket->timer->stop();
55  listenSocket->timer->close();
56  listenSocket->timer = nullptr;
57 
58  listenSocket->setCb(accept_poll_cb<A>);
59  listenSocket->start(listenSocket->nodeData->loop, listenSocket, UV_READABLE);
60  }
61  do {
62  SSL *ssl = nullptr;
63  if (listenSocket->sslContext) {
64  ssl = SSL_new(listenSocket->sslContext.getNativeContext());
65  SSL_set_accept_state(ssl);
66  }
67 
68  Socket *socket = new Socket(listenSocket->nodeData, listenSocket->nodeData->loop, clientFd, ssl);
69  socket->setPoll(UV_READABLE);
70  A(socket);
71  } while ((clientFd = netContext->acceptSocket(serverFd)) != INVALID_SOCKET);
72  }
